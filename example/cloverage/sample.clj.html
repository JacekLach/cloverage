068  (defn palindrome?
069    "Tests whether s is a palindrom."
070    ;; fully covered
071    [s]
072    (if-not (vector? s)
073      (palindrome? (vec s))
074      (if (<= (count s) 1)
075        true
076        (and (= (s 0) (s (dec (count s))))
077             (palindrome? (subvec s 1 (dec (count s))))))))
078  
079  (deftest test-palindrome
080    ;; Palindrome is fully covered
081    (is (palindrome? "noon"))
082    (is (palindrome? "racecar"))
083    (is (not (palindrome? "hello"))))

085  (defn permutation?
086    "Tests whether a and b are permutations of each other"
087    [a b]
088    (and (= (count a)
089            (count b))
090         (let [add-occurrence (fn [m c] (assoc m c (inc (m c 0))))
091               a-counts (reduce add-occurrence {} a)
092               b-counts (reduce add-occurrence {} b)]
093           (= a-counts b-counts))))
094  
095  (deftest test-permutation
096    ;; permutation is partially covered
097    (is (not (permutation? "foo" "foobar"))))
